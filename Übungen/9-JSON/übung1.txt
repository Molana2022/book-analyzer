// a) Definition des Arrays von Studentenobjekten 
let studenten = [ 
    {name: "Max Mustermann", fachrichtung: "Informatik", semester: 1}, 
    {name: "Erika Musterfrau", fachrichtung: "Maschinenbau", semester: 3} 
]; 

// b) Hinzufügen von zwei neuen Studenten 

studenten.push({name: "Anna Beispiel", fachrichtung: "Informatik", semester: 1}); 

studenten.push({name: "Peter Beispielhaft", fachrichtung: "Elektrotechnik", semester: 2}); 

 

// c) Funktion zur Ausgabe von Informatik-Studenten im ersten Semester 

function informatikErstsemester(studentenArray) { 
    studentenArray.forEach(student => { 
        if(student.fachrichtung === "Informatik" && student.semester === 1) { 
            console.log(student.name); 
        } 
    }); 
} 
 

// d) Erstellung eines neuen Arrays mit den Namen der Studenten 

let studentenNamen = studenten.map(student => student.name); 

 

// e) Konvertierung des Arrays in einen JSON-String 

let studentenJSON = JSON.stringify(studenten); 

 

// f) Parsing des JSON-Strings zurück in ein Array von Objekten 

let studentenArray = JSON.parse(studentenJSON); 

 

// g) Ausgabe des resultierenden Arrays 

console.log(studentenArray); 

 

// Aufruf der Funktion aus c) 

informatikErstsemester(studenten); 

 

// Ausgabe des Arrays mit den Namen der Studenten 

console.log(studentenNamen);
